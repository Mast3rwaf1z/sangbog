\begin{sang}{Vi har lejrbål her}{}
\begin{vers}
Vi har lejrbål her
mellem stammer og træer
på den lille ø
i den blanke sø.
\end{vers}
\begin{vers}
Det er Mexicos land
og en modig mand
han ka´ klare sig der
med et godt gevær.
\end{vers}
\begin{vers}
Men på Danger Hill
venter Sorte Bill
Ved et røveri
har han myrdet ti.
\end{vers}
\begin{vers}
Vor sherif er død
Det var Bill, der skød,
og hvor blodet flød,
dér blev jorden rød.
\end{vers}
\begin{vers}
Vi har fanget Bill
ham fra Danger Hill
og hans usle krop
den skal klynges op.
\end{vers}
\begin{vers}
Så nu ka vi bo
her i fred og ro
på den lille ø
i den blanke sø.
\end{vers}
\end{sang}
